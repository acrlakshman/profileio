Primary scalars/fields in this section include

basics:
# required
name:
...
# required
label:
...
# required
image:
...
# required
email:
...
# optional
phone:
...
# required
url:
...
# required
# (For an example of how this is rendered, please take a look at https://profileio.lakshmananumolu.com/)
summary:
...
# optional
location:
...
# optional
profiles:
...

basics:
# required
name:
value: Lakshmi Vera

basics:
# required
label:
value: Senior Software Engineer
render: true

basics:
# required
image:
value: pic.png
render: true

basics:
# required
email:
value: anumolu@uwalumni.com

basics:
# optional
phone:
value: '' # string (if you want to just enter numbers, enclose it in quotes)
render: false

url:#

basics:
# required
url:
value: 'https://profileio.lakshmananumolu.com'
render: false

summary:#

basics:
# required
# (For an example of how this is rendered, please take a look at https://profileio.lakshmananumolu.com/)
summary:
label: About me
value: summary.md # Enter the markdown content here or in a file. File path is relative to `_profile/`
render: true

basics:
# optional

value:
address: Street Name
postalCode: WI XXXXX
city: Madison
countryName: USA
region: Wisconsin
render: false

  • "github"

  • "linkedin"
  • "youtube"
  • "vimeo"
  • "twitter"
  • "facebook"
  • "twitch"
